```typescript
// BattleManager.ts 优化
export class BattleManager {
    // 新增：智能目标选择
    private _selectTarget(attacker: Pet, enemies: Enemy[]): Enemy {
        // 优先选择被克制的敌人
        const counterTargets = enemies.filter(e => 
            this._checkElementBonus(attacker.element, e.element) > 1
        );
        
        if (counterTargets.length > 0) {
            // 选择血量最低的
            return counterTargets.sort((a, b) => a.hp - b.hp)[0];
        }
        
        // 默认选择血量最低的
        return enemies.filter(e => e.hp > 0).sort((a, b) => a.hp - b.hp)[0];
    }
    
    // 新增：连击系统
    private _comboCount: number = 0;
    private _comboMultiplier: number = 1;
    
    private _updateCombo(): void {
        this._comboCount++;
        this._comboMultiplier = 1 + Math.min(this._comboCount * 0.05, 0.5);
    }
}
```

| 维度 | 评分 | 说明 |

4. **变现深度不足** - 付费点设计不够吸引
